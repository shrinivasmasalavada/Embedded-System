int soil = 34;
int relay = 2;

void setup() {
  pinMode(relay, OUTPUT);
  Serial.begin(115200);
}

void loop() {
  int moisture = analogRead(soil);

  Serial.println(moisture);

  if(moisture > 3000) {
    digitalWrite(relay, HIGH);
  } else {
    digitalWrite(relay, LOW);
  }

  delay(1000);
}
